\begin{theorem}[The deeper sector selects the phase-neutral minimizing branch]
\label{thm:neutralselection}
The minimizers of the deeper energy \eqref{eq:deepenergy} are exactly
\begin{equation}
\RespDeepDensity(x)\equiv \RespDeepDensityEq,
\qquad
\RespDeepAux(x)\equiv0,
\qquad

\label{eq:deepground}
\end{equation}
for constant \(\RespOrderDir\in\mathbb S^1\). Equivalently the zero-current phase-neutral branch is the unique minimizing branch modulo the global \(O(2)\) phase symmetry.
\end{theorem}

\begin{proof}

\qquad

\qquad

\qquad
\RespDeepAux(x)=0
\label{eq:minconditions}
\end{equation}
for all \(x\). The last two identities imply
\begin{equation}
\RespOrderCurrent(x)=\RespDeepDensity(x)\RespOrderPhase'(x)=0.
\label{eq:currentzero}
\end{equation}
Since \(\RespDeepDensity(x)=\RespDeepDensityEq>0\), equation \eqref{eq:currentzero} forces \(\RespOrderPhase'(x)=0\), so \(\RespOrderPhase\) is constant.
